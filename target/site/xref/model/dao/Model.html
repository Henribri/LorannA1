204 		//Check bump between element
205 		public Permeability checkBump(int x, int y) {
206 			
207 			Permeability permBump = null;
208 			for (IElement obj : Arimages) {
209 
210 				if ((obj.getX() == x) && (obj.getY() == y)) {
211 					if ((obj.getPerm() == Permeability.COLLECTABLE)) {
212 						this.permBump = Permeability.COLLECTABLE;						
213 					} else if (obj.getPerm() == Permeability.PENETRABLE) {
214 						this.permBump = Permeability.PENETRABLE;
215 					} else if (obj.getPerm() == Permeability.BLOCKING) {
216 						this.permBump = Permeability.BLOCKING;
217 					}
218 				}
219 			}
220 			for (IMobile obj2 : Armobile) {
221 				if ((obj2.getX() == x) && (obj2.getY() == y)) {
222 					if (obj2.getPerm() == Permeability.MONSTER) {
223 						this.permBump = Permeability.MONSTER;
224 					} else if (obj2.getPerm() == Permeability.CHARACTER) {
225 						this.permBump = Permeability.CHARACTER;
226 					} else if (obj2.getPerm() == Permeability.SPELL) {
227 						this.permBump = Permeability.SPELL;
228 					}
229 				}
230 			}
231 			return this.permBump;
232 		}
